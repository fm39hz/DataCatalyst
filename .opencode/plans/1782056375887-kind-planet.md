## 1. Unity — MonoBehaviour + Materializer

```csharp
using DataCatalyst;
using DataCatalyst.Abstractions;
using DataCatalyst.Core;
using DataCatalyst.Loaders;
using DataCatalyst.Extensions.Materialization;
using DataCatalyst.Plugins.GameConcept;
using DataCatalyst.Plugins.StateEngine;
using DataCatalyst.Plugins.StateEngine.Core;
using System.Text.Json;
using UnityEngine;

public class GameBootstrap : MonoBehaviour {
    public static DataCatalog Catalog { get; private set; }
    public static BakedStateGroup DragonAI { get; private set; }

    // Materializers per target type
    public static DataMaterializer<GameObject> EnemyMaterializer { get; private set; }
    public static DataMaterializer<GameObject> ItemMaterializer { get; private set; }

    void Awake() {
        var env = new DataCatalystEnvironment();
        env.Plugins.Register<GameConceptPlugin>();

        var options = new JsonSerializerOptions { TypeInfoResolver = new DefaultJsonTypeInfoResolver() };
        var loaded = JsonDataLoader.LoadDirectory(
            Path.Combine(Application.streamingAssetsPath, "Data"), options, env);
        var graph = DataGraphBuilder.Build(loaded.Entries, env: env);
        Catalog = DataCatalogBuilder.Resolve(graph, env: env);

        // Pre-bake state machines
        DragonAI = StateEngineBaker.Bake(
            Catalog.Get<StateGroup>(Concept.CombatAI.CombatAI), Catalog);

        // Setup materializers
        EnemyMaterializer = new DataMaterializer<GameObject>();
        EnemyMaterializer.Register<Health>((go, c) => {
            var h = go.GetComponent<HealthBar>();
            h.Max = c.Current; h.Current = c.Current;
        });
        EnemyMaterializer.Register<DamageBreath>((go, d) =>
            go.GetComponent<BreathWeapon>().Init(d.Type, d.Min, d.Max, d.Cooldown));
        EnemyMaterializer.Register<Loot>((go, l) =>
            go.GetComponent<LootDrop>().SetGold(l.Gold.Min, l.Gold.Max));

        ItemMaterializer = new DataMaterializer<GameObject>();
        ItemMaterializer.Register<Heal>((go, h) =>
            go.GetComponent<Potion>().HealAmount = h.Instant);
    }

    void SpawnDragon(string entryKey) {
        var entry = Catalog.Entries[entryKey];
        var prefab = Resources.Load<GameObject>("Dragon");
        var go = Instantiate(prefab);
        EnemyMaterializer.Materialize(entry, go);

        // Gán state machine cho dragon
        var ai = go.GetComponent<DragonAI>();
        ai.baked = DragonAI;
        ai.viableStates = new HashSet<int> { 0, 1, 2, 3 }; // Idle, Fly, BreathAttack, ClawAttack
        ai.currentState = DragonAI.DefaultStateId;
    }

    void SpawnItem(string entryKey) {
        var entry = Catalog.Entries[entryKey];
        var prefab = Resources.Load<GameObject>("ItemPickup");
        var go = Instantiate(prefab);
        ItemMaterializer.Materialize(entry, go);
        go.GetComponent<ItemPickup>().entryKey = entryKey;
    }
}

public class DragonAI : MonoBehaviour {
    public BakedStateGroup baked;
    public HashSet<int> viableStates;
    public int currentState;

    float breathCooldownTimer;
    Transform player;

    void Update() {
        var result = StateEngineEvaluator.Evaluate(
            currentState, baked, viableStates,
            signalId => signalId switch {
                0 => Vector3.Distance(transform.position, player.position), // PlayerDistance
                1 => breathCooldownTimer,  // BreathCooldown
                _ => 0f
            });

        if (result.HasValue && result.TargetStateId != currentState) {
            currentState = result.TargetStateId;
            OnStateEnter(currentState);
        }

        if (breathCooldownTimer > 0) breathCooldownTimer -= Time.deltaTime;
    }

    void OnStateEnter(int state) {
        if (state == 2) { // BreathAttack
            breathCooldownTimer = Catalog.Get<DamageBreath>(Concept.Enemy.FireDragon).Cooldown;
            // Spawn projectile
        }
    }
}

public class PlayerInventory : MonoBehaviour {
    DataMaterializer<PlayerInventory> itemApplier;

    void Awake() {
        itemApplier = new DataMaterializer<PlayerInventory>();
        itemApplier.Register<Heal>((inv, h) => inv.HealPlayer(h.Instant));
        itemApplier.Register<ManaRestore>((inv, m) => inv.RestoreMana(m.Value));
    }

    public void UseItem(string itemKey) {
        itemApplier.Materialize(GameBootstrap.Catalog.Entries[itemKey], this);
    }

    void HealPlayer(int amount) { /* ... */ }
    void RestoreMana(int amount) { /* ... */ }
}
```

## 2. Unity — DOTS (1 System xử lý tất cả AI)

```csharp
using Unity.Entities;
using Unity.Burst;
using DataCatalyst.Core;
using DataCatalyst.Plugins.StateEngine.Core;

// DC structs → ECS IComponentData
public struct AIStateECS : IComponentData { public int CurrentState; }
public struct AIOwner  : IComponentData { public Entity Owner; }
public struct BreathTimer : IComponentData { public float Remaining; }

[BurstCompile]
public partial struct AISystem : ISystem {
    private BakedStateGroup baked;   // 1 baked group cho tất cả dragons
    private float playerDistance;

    public void OnCreate(ref SystemState state) {
        baked = GameBootstrap.DragonAI;  // bake 1 lần
    }

    public void OnUpdate(ref SystemState state) {
        playerDistance = SystemAPI.GetComponent<PlayerPosition>(SystemAPI.GetSingletonEntity<PlayerTag>()).Distance;

        // 1 query quét tất cả entities có AI
        foreach (var (ai, breath) in SystemAPI.Query<RefRW<AIStateECS>, RefRW<BreathTimer>>()) {
            var result = StateEngineEvaluator.Evaluate(
                ai.ValueRW.CurrentState, baked, null, ReadSensor);

            if (result.HasValue) {
                ai.ValueRW.CurrentState = result.TargetStateId;
                if (result.TargetStateId == 2)  // BreathAttack
                    breath.ValueRW.Remaining = 5f;  // cooldown 5s
            }
        }
    }

    // readSensor = closure trên component data, không per-entity
    float ReadSensor(int signalId) => signalId switch {
        0 => playerDistance,
        1 => 0f, // BreathCooldown — đọc từ component timer
        _ => 0f
    };
}
```

## 3. Godot — Node + Materializer + StateEngine

```csharp
using Godot;
using DataCatalyst;
using DataCatalyst.Abstractions;
using DataCatalyst.Core;
using DataCatalyst.Loaders;
using DataCatalyst.Extensions.Materialization;
using DataCatalyst.Plugins.GameConcept;
using DataCatalyst.Plugins.StateEngine.Core;
using System.Text.Json;

public partial class GameManager : Node {
    public static DataCatalog Catalog;
    public static DataMaterializer<CharacterBody3D> EnemyMat;
    static BakedStateGroup dragonAI;

    public override void _Ready() {
        var env = new DataCatalystEnvironment();
        env.Plugins.Register<GameConceptPlugin>();

        var json = JsonDataLoader.LoadDirectory("res://Data", JsonOptions.Default, env);
        var graph = DataGraphBuilder.Build(json.Entries, env: env);
        Catalog = DataCatalogBuilder.Resolve(graph, env: env);

        dragonAI = StateEngineBaker.Bake(
            Catalog.Get<StateGroup>(Concept.CombatAI.CombatAI), Catalog);

        EnemyMat = new DataMaterializer<CharacterBody3D>();
        EnemyMat.Register<Health>((node, h) => {
            var hp = node.GetNode<HealthBar>("HealthBar");
            hp.MaxValue = h.Current; hp.Value = h.Current;
        });
        EnemyMat.Register<Loot>((node, l) => node.SetMeta("GoldMin", l.Gold.Min));
    }

    public static void SpawnDragon(string key, Node3D parent) {
        var scene = GD.Load<PackedScene>("res://Enemies/Dragon.tscn");
        var dragon = scene.Instantiate<CharacterBody3D>();
        EnemyMat.Materialize(Catalog.Entries[key], dragon);
        parent.AddChild(dragon);

        var ai = new DragonAIComponent();
        ai.Baked = dragonAI;
        ai.CurrentState = dragonAI.DefaultStateId;
        ai.ViableStates = new HashSet<int> { 0, 1, 2, 3 };
        dragon.AddChild(ai);
    }
}

public partial class DragonAIComponent : Node {
    public BakedStateGroup Baked;
    public int CurrentState;
    public HashSet<int> ViableStates;
    Node3D player;

    public override void _Process(double delta) {
        var result = StateEngineEvaluator.Evaluate(
            CurrentState, Baked, ViableStates,
            id => id switch {
                0 => GlobalPosition.DistanceTo(player.GlobalPosition),
                1 => GetNode<Timer>("BreathTimer").TimeLeft,
                _ => 0f
            });

        if (result.HasValue) {
            CurrentState = result.TargetStateId;
            TransitionTo((StateType)result.TargetStateId);
        }
    }
}
```

## 4. Godot — Friflo ECS (1 System xử lý tất cả AI)

```csharp
using Friflo.Engine.ECS;

public struct AIStateECS : IComponent { public int CurrentState; }
public struct BreathTimerECS : IComponent { public float Remaining; }

public partial class EcsAISystem : Node {
    EntityStore store;
    BakedStateGroup baked;

    public override void _Ready() {
        store = GetNode<EcsBootstrap>("/root/EcsBootstrap").Store;
        baked = GameManager.DragonAI;  // 1 baked group, dùng chung
    }

    public override void _Process(double delta) {
        float playerDist = GetPlayerDistance();

        // 1 query duy nhất, evaluate tất cả
        var query = store.Query<AIStateECS, BreathTimerECS>();
        query.ForEachEntity((ref AIStateECS ai, ref BreathTimerECS bt) => {
            var result = StateEngineEvaluator.Evaluate(
                ai.CurrentState, baked, null,
                signalId => signalId switch {
                    0 => playerDist,
                    1 => bt.Remaining,
                    _ => 0f
                });

            if (result.HasValue) ai.CurrentState = result.TargetStateId;
            if (bt.Remaining > 0) bt.Remaining -= (float)delta;
        });
    }
}
```

## 5. MonoGame — Game Component + Materializer

```csharp
using Microsoft.Xna.Framework;
using DataCatalyst;
using DataCatalyst.Core;
using DataCatalyst.Loaders;
using DataCatalyst.Extensions.Materialization;
using DataCatalyst.Plugins.StateEngine.Core;

public class DragonGame : Game {
    DataCatalog catalog;
    DataMaterializer<DragonEntity> materializer;
    BakedStateGroup bakedAI;
    List<DragonEntity> dragons = new();

    protected override void LoadContent() {
        var env = new DataCatalystEnvironment();
        env.Plugins.Register<GameConceptPlugin>();
        var json = JsonDataLoader.LoadDirectory("Content/Data", JsonOptions.Default, env);
        var graph = DataGraphBuilder.Build(json.Entries, env: env);
        catalog = DataCatalogBuilder.Resolve(graph, env: env);
        bakedAI = StateEngineBaker.Bake(
            catalog.Get<StateGroup>(Concept.CombatAI.CombatAI), catalog);

        materializer = new DataMaterializer<DragonEntity>();
        materializer.Register<Health>((d, h) => { d.HP = h.Current; d.MaxHP = h.Max; });
        materializer.Register<DamageBreath>((d, b) => { d.BreathDmg = b.Min; d.BreathCooldown = b.Cooldown; });
        materializer.Register<Loot>((d, l) => d.GoldDrop = (l.Gold.Min, l.Gold.Max));

        // Eager load
        foreach (var key in new[] { "FireDragon", "IceDragon", "Dragon" }) {
            var e = new DragonEntity { Id = key, AIState = bakedAI.DefaultStateId };
            materializer.Materialize(catalog.Entries[key], e);
            dragons.Add(e);
        }
    }

    protected override void Update(GameTime gt) {
        foreach (var d in dragons) {
            var result = StateEngineEvaluator.Evaluate(
                d.AIState, bakedAI, null,
                id => ReadSensor(d, id));
            if (result.HasValue) d.AIState = result.TargetStateId;
        }
    }
}

public class DragonEntity {
    public string Id;
    public int HP, MaxHP;
    public int BreathDmg, GoldDrop;
    public float BreathCooldown;
    public int AIState;
    public Vector2 Position;
}
```

## 6. MonoGame — Friflo ECS (1 System xử lý tất cả AI)

```csharp
using Friflo.Engine.ECS;

public struct AIStateECS     : IComponent { public int CurrentState; }
public struct BreathTimerECS : IComponent { public float Remaining; }

public class AISystem {
    EntityStore store;
    BakedStateGroup baked;

    public AISystem(EntityStore store, BakedStateGroup baked) {
        this.store = store;
        this.baked = baked;  // 1 baked group, dùng chung
    }

    public void Update(float playerDistance, float delta) {
        var query = store.Query<AIStateECS, BreathTimerECS>();
        query.ForEachEntity((ref AIStateECS ai, ref BreathTimerECS bt) => {
            var result = StateEngineEvaluator.Evaluate(
                ai.CurrentState, baked, null,
                signalId => signalId switch {
                    0 => playerDistance,
                    1 => bt.Remaining,
                    _ => 0f
                });

            if (result.HasValue) ai.CurrentState = result.TargetStateId;
            if (bt.Remaining > 0) bt.Remaining -= delta;
        });
    }
}
```

## Pattern Summary

| Pattern                | How DC handles it                              | Consumer code                                              |
| ---------------------- | ---------------------------------------------- | ---------------------------------------------------------- |
| **Deeply nested data** | SourceGen generates struct per nesting level   | `catalog.Get<DamageBreath>(Concept.Enemy.FireDragon).Type` |
| **Inheritance**        | ComponentMerger merges parent→child at resolve | Same API — inheritance invisible                           |
| **Multi-concept**      | Entry belongs to N concepts                    | Accessible via any `Concept.X.Y`                           |
| **Materializer**       | Register mapping once, apply anywhere          | `mat.Register<T>((obj, c) => assign(obj, c))`              |
| **StateEngine**        | Pre-bake at load, evaluate per frame           | `Evaluate(current, baked, viable, readSensor)`             |
| **AISensor-driven**    | Sensor = entry with "AISensor" concept         | `readSensor` lambda, signalId = entry ID                   |
| **ECS bridge**         | DC → ECS at init                               | ForEach over store.Query()                                 |
| **Immutable catalog**  | Read-only after Resolve                        | No mutation API on DataCatalog                             |
